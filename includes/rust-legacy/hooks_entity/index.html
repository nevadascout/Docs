Entity Hooks

OnAirdrop

void OnAirdrop(Vector3 position)
{
    Puts("OnAirdrop works!");
}

PLUGIN:OnAirdrop(position)
    print("OnAirdrop works!")
end

OnAirdrop: (position) =>
  print "OnAirdrop works!"

OnAirdrop: function(position) {
  print("OnAirdrop works!");
}

def OnAirDrop(self, position):
  print "OnAirDrop works!"
  • Called when an airdrop has been called
  • No return behavior
